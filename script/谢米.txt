
#在谢米背后进入地下洞窟，进秘密基地墙边测帧，到闪帧运行脚本
_闪帧 = 60             #   在这里输入你距离闪帧还差多少，从运行脚本开始
_等待时间 = 8

# 唤醒设备
FOR 5
    A 120
    100
NEXT

$2 = _闪帧
$3 = _等待时间
$3 *= 1000

IF $2 < 0
    $2 = 1
ENDIF
FOR $2
    1010
NEXT
B 120
LS DOWN,1200
WAIT 2700
Y 200
WAIT 400
DOWN 200
WAIT 200
FOR 80
    A 20
    WAIT 10
NEXT
WAIT 1500
FOR 240
    A 20
    WAIT 10
NEXT
WAIT $3
FOR 8
    A 200
NEXT
